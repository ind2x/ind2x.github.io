#include <iostream>
#include <vector>
using namespace std;
typedef unsigned long long int uli;

uli ans[1000001]={};

int main() {
	ios::sync_with_stdio(false); 
    cin.tie(NULL);
	
	uli s,b, cnt=0;
	cin >> s >> b; /* 1 1000001000000 -> 에러발생함 ... */
	uli size = (b < 1000000) ? b+1 : 1000001; 
	/* 최대 10^6까지의 소수만 구해야 함*/ 
	/* 10^6을 넘어가는 소수를 구해도 그 제곱수는 MAX를 넘어가버리기 때문*/
	vector<bool> check(size, true);
	for(uli i=2; i<size; i++) 
	{
		if(check[i]) /* 소수일 떄*/
		{
			for(uli j=i*2; j < size; j+=i) 
			{
				check[j]=false; /* 에라토스테네스의 체 */
			}

			uli k=s/(i*i); /* k=10/4=2*/
			/* 
				굳이 4*1부터 구할 필요가 없었음.
				예를 들어 s가 10, b=20 이라면
				10 ~ 20 사이의 제곱 ㄴㄴ 수를 구하려면
				10보다 크고 20보다 작은 4의 배수를 구하면 되는 것임.
				또한 12가 제곱ㅇㅇ수라면 ans[12]에 저장하는 것이 아니라
				ans[12-10]=ans[2]에 저장함으로써 할당 가능한 범위 내에서 구분 가능.
			*/
			if(s%(i*i) != 0) { k++;} /* 10%4=2 -> 4*k < 10 -> 4*(k+1) > 10 */ \
			/*
			for(uli j=(i*i)*k; j <= b; j+=i*i) 
			{
				ans[j-s]=1;
			}
			*/
			while(i*i*k <= b) {
				ans[(i*i*k)-s]=1;
				k++;
			}
		}
	}
	for(uli i=0; i<= b-s; i++) 
	/*
		ans[0] == ans[10]임.
		즉, 10부터 20 사이의 제곱 ㄴㄴ 수를 구한다면
		ans[0] 부터 ans[b-s]에 
	*/
	{
		if(ans[i] != 1) { cnt++; }
	}

	cout << cnt << '\n';
	return 0;
}
